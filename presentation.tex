% Top part (Page count and section names)
\setbeamertemplate{headline}{%
	\begin{beamercolorbox}[ht=4.5ex,dp=0ex]{headline}
		% navigation
		\strut~ \insertshorttitle~%
		\setbox0=\hbox{\insertsection}\ifdim\wd0=0pt\else~{\footnotesize ›}~~\insertsection\fi~%
		\setbox0=\hbox{\insertsubsection}\ifdim\wd0=0pt\else~{\footnotesize ›}~~\insertsubsection\fi~%
		% page no.
		\hfill\insertframenumber{} %
		of \inserttotalframenumber {} ~\vskip2pt
	\end{beamercolorbox}
	% light blue bar
	\begin{beamercolorbox}{lower separation line head}
		\vskip3pt~
	\end{beamercolorbox}
}
% Bottom part (color bars)
\setbeamertemplate{footline}{%
	% light blue bar
	\begin{beamercolorbox}{upper separation line foot}
		\vskip3pt~
	\end{beamercolorbox}
	% dark blue bar
	\begin{beamercolorbox}{footline}
		\vspace{0.5em}
	\end{beamercolorbox}
}
% Section headings (red on light blue)
\setbeamertemplate{frametitle}{
	\ifbeamercolorempty[bg]{frametitle}{}{\nointerlineskip}%
	\begin{beamercolorbox}[leftskip=0.4cm,sep=0cm,wd=\paperwidth]{frametitle}
		\usebeamerfont{frametitle}%
		\vbox{}\vskip-1ex%
		\strut\insertframetitle\strut\par%
		{%
			\ifx\insertframesubtitle\@empty%
			\else%
			{\usebeamerfont{framesubtitle}\usebeamercolor[fg]{framesubtitle}\insertframesubtitle\strut\par}%
			\fi
		}%
		\vskip-.4cm% set inside beamercolorbox... evil here...
	\end{beamercolorbox}%
}
